\\\centerline{\Large Вариант 1.}
	\begin{enumerate}
	\item 	(1б.) Дан функционал 
		$	\displaystyle V\left[y\right]=\int\limits_0^1{x^4y'^2}dx.
		$ 
		\begin{enumerate}
			\item Найти его значение при $y(x) = x$, $y(x) = 1+x$, $y(x) = 1-x$.
			\item Достигается ли экстремум $V[y]$ на множестве функций $y(x) = kx+b$?
			Если достигается, то при каких значениях $k$ и $b$? Максимум или минимум?
		\end{enumerate}
	\item 	(1б.) Дан функционал 
		$
			\displaystyle V\left[y\right]=\int\limits_0^1
			{\left[y'^2 + y^2\right]}dx,
		$ 
		и краевые условия $y(0) = 1$, $y(1) = 2$. Пусть экстремум функционала
		ищется в классе функций $\displaystyle y(x) = \varphi(x) + C\sin{\pi x}$, где
		$C$ --- параметр, $\varphi(x) \in C^1_{[0,1]}$.
		\begin{enumerate}
			\item Подберите $\varphi(x)$ так, чтобы при любых значениях $C$
				функция $y(x)$ удовлетворяла бы краевым условиям.
			\item Достигается ли экстремум $V[y]$ на получившимся множестве
				функций $y(x)$? Если достигается, то при каких значениях 
				$C$? Максимум или минимум?
		\end{enumerate}
	\item 	(1б.) Найдите приближенное решение вариационной задачи
		\[
			\displaystyle V\left[y\right] = \int\limits_0^1
			{\left[y'^2 - y^2 - 2xy\right]}dx \to \text{extr}, \quad
			y(0) = y(1) = 0
		\]
		в виде $y(x) = x(1-x)(C_0 + C_1x + C_2x^2 + \dots + C_nx^n)$, ограничиться случаем $n=0$.
	\item 	(ДЗ 1б.) Найти приближенное решение предыдущей задачи при $n=1$ и $n=2$.
	\item[\addtocounter{enumi}{1}\theenumi*] (ДЗ 3б.) Найти приближенное решение вариационной задачи
		\[
			V\left[z(x,y)\right] = \int\limits_D\!\!\!\int 
			\left[\left(\frac{\partial z}{\partial x}\right)^2 + 
			\left(\frac{\partial z}{\partial y}\right)^2 - 2z\right]dxdy \to \text{extr},
		\]
		где область интегрирования $D = \left\{(x,y) : -1 \leqslant x \leqslant 1, 
		-1 \leqslant y \leqslant 1\right\}$ и на границе этой области $z(x,y)$ равна нулю.
		{\it Подсказка: можно попробовать искать решение в виде $z(x,y) = C(x^2-1)(y^2-1)$.}
	\end{enumerate}
